// validateCategoryField returns an inline validation indicator for range fields (temperature, top_p).
// Validation runs on field-exit (focused=false). For color, runs on every keystroke.
func validateCategoryField(label, value string, focused bool) string {
	switch label {
	case "color":
		if value == "" {
			return ""
		}
		if !wizCatValidationHexRe.MatchString(value) {
			return wizCatErrorStyle.Render(" ✗ invalid hex")
		}
		return wizCatValidStyle.Render(" ✓")
	case "temperature":
		if value == "" {
			return ""
		}
		if focused {
			return ""
		}
		if v, err := strconv.ParseFloat(value, 64); err != nil || math.IsNaN(v) || v < 0 || v > 2 {
			return wizCatErrorStyle.Render(" ✗ must be 0-2")
		}
		return wizCatValidStyle.Render(" ✓")
	case "top_p":
		if value == "" {
			return ""
		}
		if focused {
			return ""
		}
		if v, err := strconv.ParseFloat(value, 64); err != nil || math.IsNaN(v) || v < 0 || v > 1 {
			return wizCatErrorStyle.Render(" ✗ must be 0-1")
		}
		return wizCatValidStyle.Render(" ✓")
	}
	return ""
}

// validateAgentField returns an inline validation indicator for the given field.
// For simple fields (color), validation runs on every keystroke (focused=true).
